139 }
140 else // Implied if(pState->mode == MODE_GENERATE_PATCH)
141 {
142 // Reading from modified ROM to patch

144 outputFileStream = fopen(pState->outputFile, "rb");
145 generatePatch(romFileStream, outputFileStream, patchFileStream, pState->patchType);
